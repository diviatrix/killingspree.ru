ЦЕЛЬ ЗАДАНИЯ

Оценить базовые структуры ООП, понимание передачи параметров и локальных/глобальных переменных.

Задание


int a = 0;
int b = 1;
int c = 2;
int[] array = { 0, 1, 2 };

void one(int a, int b, int c) 
{
    a = 1;
    b = 2;
    c = a + b;
}

void two(int[] _array) 
{
    _array[a] = c;
    _array[b] = a;
    _array[c] = b;
}

one(a, b, c);
two(array);

Console.WriteLine(array[0] + " " + array[1] + " " + array[2]);
Console.WriteLine(a + " " + b + " " + c);

Какой будет вывод в консоли?
